import anthropic

def review(code: str) -> str:

## 5.2 システム全体像

AIコードレビューシステムの基本フローは次のとおりだ。

```
GitHub PR オープン
    ↓
GitHub Webhook → Lambda/Cloud Functions
    ↓
GitHub API でdiff取得
    ↓
Claude API でレビュー実行
    ↓
GitHub API でPRコメント投稿
```

このフローを実装するには3つのAPIを扱う必要がある。GitHub REST API（diff取得・コメント投稿）、Claude API（レビュー実行）、そして任意のオーケストレーション基盤（GitHub ActionsまたはWebhookサーバー）だ。

本章では GitHub Actions をトリガーとして使い、Python スクリプトでレビューを実行するシンプルな構成を採用する。本番稼働させるのに十分な実用性がありながら、動作を把握しやすい。

## 5.3 GitHub PRのdiffを取得する

まず GitHub API からdiffを取得する部分を実装する。

```python
# github_client.py
import os
import re
from dataclasses import dataclass
from typing import Optional

import requests


@dataclass
class PullRequestDiff:
    pr_number: int
    title: str
    body: str
    diff_text: str
    changed_files: list[str]
    additions: int
    deletions: int


class GitHubClient:
    """GitHub REST API クライアント."""

    def __init__(self, token: str, repo: str):
        """
        Args:
            token: GitHub Personal Access Token (repo スコープ必要)
            repo:  "owner/repo" 形式
        """
        self.token = token
        self.repo = repo
        self.base_url = "https://api.github.com"
        self.session = requests.Session()
        self.session.headers.update(
            {
                "Authorization": f"token {token}",
                "Accept": "application/vnd.github.v3+json",
                "X-GitHub-Api-Version": "2022-11-28",
            }
        )

    def get_pull_request(self, pr_number: int) -> dict:
        """PRのメタデータを取得する."""
        url = f"{self.base_url}/repos/{self.repo}/pulls/{pr_number}"
        resp = self.session.get(url, timeout=30)
        resp.raise_for_status()
        return resp.json()

    def get_pull_request_diff(self, pr_number: int) -> str:
        """PRのunified diffを取得する."""
        url = f"{self.base_url}/repos/{self.repo}/pulls/{pr_number}"
        resp = self.session.get(
            url,
            headers={"Accept": "application/vnd.github.v3.diff"},
            timeout=30,
        )
        resp.raise_for_status()
        return resp.text

    def get_pull_request_files(self, pr_number: int) -> list[dict]:
        """変更ファイルの一覧を取得する."""
        url = f"{self.base_url}/repos/{self.repo}/pulls/{pr_number}/files"
        files = []
        page = 1
        while True:
            resp = self.session.get(
                url, params={"per_page": 100, "page": page}, timeout=30
            )
            resp.raise_for_status()
            data = resp.json()
            if not data:
                break
            files.extend(data)
            page += 1
        return files

    def fetch_pr_diff(self, pr_number: int) -> PullRequestDiff:
        """PRのdiff情報を一括取得する."""
        pr_data = self.get_pull_request(pr_number)
        diff_text = self.get_pull_request_diff(pr_number)
        files = self.get_pull_request_files(pr_number)

        return PullRequestDiff(
            pr_number=pr_number,
            title=pr_data["title"],
            body=pr_data.get("body") or "",
            diff_text=diff_text,
            changed_files=[f["filename"] for f in files],
            additions=pr_data["additions"],
            deletions=pr_data["deletions"],
        )

    def post_review_comment(
        self,
        pr_number: int,
        body: str,
        commit_id: Optional[str] = None,
        path: Optional[str] = None,
        line: Optional[int] = None,
    ) -> dict:
        """PRにレビューコメントを投稿する.

        path/lineを指定するとインラインコメントになる。
        指定しない場合はPR全体へのコメントになる。
        """
        if path and line and commit_id:
            # インラインコメント
            url = f"{self.base_url}/repos/{self.repo}/pulls/{pr_number}/comments"
            payload = {
                "body": body,
                "commit_id": commit_id,
                "path": path,
                "line": line,
            }
        else:
            # PRへの通常コメント
            url = (
                f"{self.base_url}/repos/{self.repo}/issues/{pr_number}/comments"
            )
            payload = {"body": body}

        resp = self.session.post(url, json=payload, timeout=30)
        resp.raise_for_status()
        return resp.json()
```

## 5.4 Claude APIでdiffをレビューする

次に取得したdiffをClaudeに渡してレビューを実行する部分を実装する。

```python
# claude_reviewer.py
import os
import time
from dataclasses import dataclass

import anthropic

from github_client import PullRequestDiff


@dataclass
class ReviewResult:
    summary: str
    issues: list[dict]
    suggestions: list[str]
    approval_recommendation: str  # "approve" | "request_changes" | "comment"
    estimated_risk: str           # "low" | "medium" | "high"


class ClaudeReviewer:
    """Claude APIを使ったコードレビュアー."""

    # トークン制限（claude-3-5-sonnet の入力上限 200k）
    MAX_DIFF_CHARS = 150_000

    def __init__(self, api_key: str, model: str = "claude-3-5-sonnet-20241022"):
        self.client = anthropic.Anthropic(api_key=api_key)
        self.model = model

    def _truncate_diff(self, diff_text: str) -> tuple[str, bool]:
        """diffが長すぎる場合に末尾を切り捨てる."""
        if len(diff_text) <= self.MAX_DIFF_CHARS:
            return diff_text, False
        truncated = diff_text[: self.MAX_DIFF_CHARS]
        # hunkの途中で切れないよう最後の完全な行まで戻す
        last_newline = truncated.rfind("\n")
        if last_newline > 0:
            truncated = truncated[:last_newline]
        return truncated + "\n\n[... diff truncated due to size ...]", True

    def _build_prompt(self, pr: PullRequestDiff) -> str:
        diff_text, was_truncated = self._truncate_diff(pr.diff_text)
        truncation_note = (
            "\n**注意: diffが長いため一部省略されています。**\n"
            if was_truncated
            else ""
        )

        return f"""あなたは経験豊富なソフトウェアエンジニアです。
以下のPull Requestをレビューしてください。

## PR情報
- タイトル: {pr.title}
- 説明: {pr.body or "(説明なし)"}
- 変更ファイル数: {len(pr.changed_files)}
- 追加行数: {pr.additions} / 削除行数: {pr.deletions}

## 変更ファイル一覧
{chr(10).join(f"- {f}" for f in pr.changed_files)}

## diff
{truncation_note}
```diff
{diff_text}
```

## レビュー観点
以下の優先度順でレビューしてください。

1. **バグ・クラッシュリスク** (最優先)
   - ヌルポインタ参照、配列範囲外アクセス
   - 例外が握りつぶされているケース
   - リソースリーク（ファイルハンドル、DB接続）

2. **セキュリティ** (高優先)
   - SQLインジェクション、XSS、コマンドインジェクション
   - 認証・認可の欠落
   - シークレット情報のハードコード

3. **設計・保守性** (中優先)
   - DRY原則違反
   - 単一責任原則の逸脱
   - 命名の不明確さ

4. **パフォーマンス** (低優先)
   - N+1クエリ問題
   - 不要なループ・計算

## 出力フォーマット
必ず以下のJSON形式で回答してください。

```json
{{
  "summary": "PRの変更内容の概要（2〜3文）",
  "issues": [
    {{
      "severity": "critical|high|medium|low",
      "file": "ファイルパス",
      "line_hint": "該当行番号または範囲（不明な場合はnull）",
      "description": "問題の説明",
      "suggestion": "修正案"
    }}
  ],
  "suggestions": ["改善提案1", "改善提案2"],
  "approval_recommendation": "approve|request_changes|comment",
  "estimated_risk": "low|medium|high"
}}
```

issuesが0件の場合は空配列 [] にしてください。
"""

    def review(self, pr: PullRequestDiff, max_retries: int = 3) -> ReviewResult:
        """PRをレビューする.

        Args:
            pr: PRのdiff情報
            max_retries: レート制限時の最大リトライ回数

        Returns:
            レビュー結果

        Raises:
            anthropic.APIError: APIエラーが回復不能な場合
        """
        prompt = self._build_prompt(pr)

        for attempt in range(max_retries):
            try:
                message = self.client.messages.create(
                    model=self.model,
                    max_tokens=4096,
                    timeout=120,  # 2分タイムアウト
                    messages=[{"role": "user", "content": prompt}],
                )
                raw_text = message.content[0].text
                return self._parse_response(raw_text)

            except anthropic.RateLimitError:
                wait = 60 * (attempt + 1)  # 60秒、120秒、180秒
                print(f"[WARN] Rate limit hit. Waiting {wait}s (attempt {attempt+1}/{max_retries})")
                time.sleep(wait)

            except anthropic.APITimeoutError:
                wait = 30 * (attempt + 1)
                print(f"[WARN] Timeout. Waiting {wait}s (attempt {attempt+1}/{max_retries})")
                time.sleep(wait)

            except anthropic.APIStatusError as e:
                if e.status_code >= 500:
                    # サーバーエラーはリトライ対象
                    wait = 30 * (attempt + 1)
                    print(f"[WARN] Server error {e.status_code}. Waiting {wait}s")
                    time.sleep(wait)
                else:
                    raise  # 4xxはリトライしない

        raise RuntimeError(f"Max retries ({max_retries}) exceeded")

    def _parse_response(self, raw_text: str) -> ReviewResult:
        """Claude の回答からJSONを抽出してパースする."""
        import json
        import re

        # コードブロック内のJSONを抽出
        json_match = re.search(r"```(?:json)?\s*(\{.*?\})\s*```", raw_text, re.DOTALL)
        if json_match:
            json_str = json_match.group(1)
        else:
            # コードブロックなしでJSONが含まれる場合
            start = raw_text.find("{")
            end = raw_text.rfind("}") + 1
            if start == -1 or end == 0:
                raise ValueError(f"JSON not found in response: {raw_text[:200]}")
            json_str = raw_text[start:end]

        data = json.loads(json_str)
        return ReviewResult(
            summary=data.get("summary", ""),
            issues=data.get("issues", []),
            suggestions=data.get("suggestions", []),
            approval_recommendation=data.get("approval_recommendation", "comment"),
            estimated_risk=data.get("estimated_risk", "medium"),
        )
```

## 5.5 レビュー結果をPRコメントとして投稿する

```python
# comment_formatter.py
from claude_reviewer import ReviewResult

SEVERITY_EMOJI = {
    "critical": "🔴",
    "high": "🟠",
    "medium": "🟡",
    "low": "🔵",
}

RECOMMENDATION_EMOJI = {
    "approve": "✅",
    "request_changes": "🚫",
    "comment": "💬",
}

RISK_EMOJI = {
    "low": "🟢",
    "medium": "🟡",
    "high": "🔴",
}


def format_review_comment(result: ReviewResult) -> str:
    """ReviewResultをMarkdownコメントに変換する."""
    rec_emoji = RECOMMENDATION_EMOJI.get(result.approval_recommendation, "💬")
    risk_emoji = RISK_EMOJI.get(result.estimated_risk, "🟡")

    lines = [
        "## 🤖 AI Code Review (Claude)",
        "",
        f"**判定:** {rec_emoji} `{result.approval_recommendation}`　"
        f"**リスク:** {risk_emoji} `{result.estimated_risk}`",
        "",
        "### 概要",
        result.summary,
        "",
    ]

    if result.issues:
        lines += ["### 指摘事項", ""]
        for i, issue in enumerate(result.issues, 1):
            emoji = SEVERITY_EMOJI.get(issue["severity"], "⚪")
            file_info = issue.get("file", "")
            line_info = issue.get("line_hint")
            location = f"`{file_info}`" if file_info else ""
            if line_info:
                location += f" L{line_info}"

            lines += [
                f"#### {emoji} [{issue['severity'].upper()}] {location}",
                "",
                issue["description"],
                "",
                f"**修正案:** {issue['suggestion']}",
                "",
            ]
    else:
        lines += ["### 指摘事項", "", "特に問題は見つかりませんでした。", ""]

    if result.suggestions:
        lines += ["### 改善提案", ""]
        for s in result.suggestions:
            lines.append(f"- {s}")
        lines.append("")

    lines += [
        "---",
        "_このコメントはClaude APIにより自動生成されました。最終判断は人間のレビュアーが行ってください。_",
    ]

    return "\n".join(lines)
```

```python
# reviewer.py  ← エントリポイント（完全動作版）
"""
使い方:
  export GITHUB_TOKEN=ghp_xxx
  export ANTHROPIC_API_KEY=sk-ant-xxx
  export GITHUB_REPOSITORY=owner/repo
  python reviewer.py --pr 42
"""
import argparse
import os
import sys

from claude_reviewer import ClaudeReviewer
from comment_formatter import format_review_comment
from github_client import GitHubClient


def main() -> None:
    parser = argparse.ArgumentParser(description="AI Code Reviewer")
    parser.add_argument("--pr", type=int, required=True, help="PR番号")
    parser.add_argument("--dry-run", action="store_true", help="コメント投稿をスキップ")
    args = parser.parse_args()

    github_token = os.environ["GITHUB_TOKEN"]
    anthropic_key = os.environ["ANTHROPIC_API_KEY"]
    repo = os.environ["GITHUB_REPOSITORY"]  # "owner/repo"

    github = GitHubClient(token=github_token, repo=repo)
    reviewer = ClaudeReviewer(api_key=anthropic_key)

    print(f"[INFO] Fetching PR #{args.pr} from {repo}...")
    pr_diff = github.fetch_pr_diff(args.pr)
    print(
        f"[INFO] PR: {pr_diff.title} (+{pr_diff.additions}/-{pr_diff.deletions} lines)"
    )

    print("[INFO] Running AI review...")
    result = reviewer.review(pr_diff)
    print(f"[INFO] Review done: {len(result.issues)} issues, risk={result.estimated_risk}")

    comment_body = format_review_comment(result)

    if args.dry_run:
        print("\n--- DRY RUN: コメント内容 ---")
        print(comment_body)
        print("--- END ---")
    else:
        print("[INFO] Posting comment to GitHub...")
        github.post_review_comment(pr_number=args.pr, body=comment_body)
        print("[INFO] Done.")

    # クリティカルな問題がある場合は終了コード1を返す
    has_critical = any(
        issue["severity"] in ("critical", "high") for issue in result.issues
    )
    if has_critical:
        print("[WARN] Critical/High severity issues found.")
        sys.exit(1)


if __name__ == "__main__":
    main()
```

## 5.6 コスト見積もり

Claude APIの課金はトークン単位で発生する。claude-3-5-sonnet の料金（2024年末時点）は入力$3/MTok、出力$15/MTokだ。PRサイズ別の概算コストを示す。

| PRサイズ | diff行数目安 | 入力トークン | 出力トークン | 1回あたりコスト |
|---------|------------|------------|------------|--------------|
| 小 (S)  | ~100行     | ~3,000     | ~800       | ~$0.02       |
| 中 (M)  | ~500行     | ~10,000    | ~1,200     | ~$0.05       |
| 大 (L)  | ~2,000行   | ~35,000    | ~1,500     | ~$0.13       |
| 特大(XL)| ~5,000行   | ~80,000    | ~2,000     | ~$0.27       |

1日10件のPRをレビューする場合、月間コストは $15〜$80 程度になる。1人のエンジニアの時給を考えると、コードレビューに費やす時間削減効果の方がはるかに大きい。

コスト削減のポイントは2つある。1つ目は**モデルの使い分け**だ。ルーティングしてテスト変更のみのPRはHaiku（$0.25/MTok）でレビューする。2つ目は**diffの絞り込み**で、自動生成ファイル（lockファイル、migrationの番号部分）を除外することで入力トークンを削減できる。

## 🚨 本番落とし穴

### 落とし穴① diffが大きすぎてコンテキスト超過する

100kトークンを超えるような巨大PRでは、全diffをそのまま渡すとコンテキスト上限に達する。前述の `_truncate_diff` は単純な切り捨てだが、より良い対策は**ファイルごとにチャンク分割して並列レビューする**方法だ。

```python
def review_by_file(self, pr: PullRequestDiff) -> list[ReviewResult]:
    """ファイル単位でレビューして結果をマージする."""
    results = []
    for file_diff in split_diff_by_file(pr.diff_text):
        result = self.review_single_file(file_diff)
        results.append(result)
    return merge_results(results)
```

### 落とし穴② GitHub APIのレート制限

GitHub APIは認証済みで5,000 req/hの制限がある。大量のPRを処理する場合は `X-RateLimit-Remaining` ヘッダーを確認し、残数が少ない場合は待機する実装が必要だ。

```python
def _check_rate_limit(self, response: requests.Response) -> None:
    remaining = int(response.headers.get("X-RateLimit-Remaining", 999))
    if remaining < 10:
        reset_at = int(response.headers.get("X-RateLimit-Reset", 0))
        wait = max(reset_at - time.time() + 5, 0)
        print(f"[WARN] Rate limit low ({remaining}). Waiting {wait:.0f}s")
        time.sleep(wait)
```

### 落とし穴③ JSONパースエラーで落ちる

Claude が必ずしも指定したJSON形式で返答するとは限らない。特に長いdiffの場合、途中で出力が打ち切られてJSON が不完全になることがある。`max_tokens` を十分大きく設定し（4096推奨）、JSONパース失敗時はフォールバックとして生テキストをそのままコメントに投稿する仕組みを入れておくとよい。

## 実務Tips

**テスト環境の構築:** `--dry-run` フラグを使えばGitHubへの書き込みなしに動作確認できる。CIでは `act` を使ってローカルでGitHub Actionsを実行すると開発速度が上がる。

**既存PRへの再レビュー:** `python reviewer.py --pr 42` をローカルから実行するだけで既存のPRに対してもレビューを実行できる。デバッグに便利だ。

**レビューのスキップ:** PRタイトルに `[skip-ai-review]` を含む場合はスキップするロジックをワークフローに追加しておくと、ドキュメント修正などのPRで不要なレビューを避けられる。

## まとめ

- GitHub PRのdiffは `application/vnd.github.v3.diff` ヘッダーで取得できる
- ClaudeへはJSON出力フォーマットを指定し、レビュー観点を優先度順に明示する
- エラーハンドリングはレート制限・タイムアウト・JSONパース失敗の3つを必ず実装する
- コストはPRサイズに比例するが、実務的には月$50〜100の範囲に収まることが多い
