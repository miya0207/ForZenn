```python
import json
from dataclasses import dataclass

@dataclass
class PeriodMetrics:
    """期間別メトリクス."""

    period_name: str
    total_prs: int
    bugs_found_in_review: int    # レビュー段階で発見されたバグ
    bugs_escaped_to_staging: int  # ステージングまで抜けたバグ
    bugs_escaped_to_prod: int    # 本番まで抜けたバグ
    total_review_hours: float    # 人間レビューの総工数（時間）
    api_cost_usd: float          # AIのAPI費用（ドル）

before = PeriodMetrics(
    period_name="導入前（2024年4-6月）",
    total_prs=312,
    bugs_found_in_review=47,
    bugs_escaped_to_staging=31,
    bugs_escaped_to_prod=12,
    total_review_hours=186.0,
    api_cost_usd=0.0,
)

after = PeriodMetrics(
    period_name="導入後（2024年7-9月）",
    total_prs=328,
    bugs_found_in_review=68,    # レビュー段階での発見が増加
    bugs_escaped_to_staging=14, # ステージングバグが減少
    bugs_escaped_to_prod=5,     # 本番バグが減少
    total_review_hours=142.0,   # 人間レビュー工数が減少
    api_cost_usd=3.82,          # 3ヶ月のAPI費用合計
)


def calculate_improvement(before: PeriodMetrics, after: PeriodMetrics) -> dict:
    """導入前後の改善率を計算する."""
    # バグ率（100 PRあたりの本番バグ数）
    before_bug_rate = (before.bugs_escaped_to_prod / before.total_prs) * 100
    after_bug_rate = (after.bugs_escaped_to_prod / after.total_prs) * 100
    bug_rate_improvement = (before_bug_rate - after_bug_rate) / before_bug_rate

    # レビュー効率（1PRあたりの工数）
    before_hours_per_pr = before.total_review_hours / before.total_prs
    after_hours_per_pr = after.total_review_hours / after.total_prs
    efficiency_improvement = (before_hours_per_pr - after_hours_per_pr) / before_hours_per_pr

    return {
        "before_bug_rate_per_100pr": round(before_bug_rate, 2),
        "after_bug_rate_per_100pr": round(after_bug_rate, 2),
        "bug_rate_improvement": f"-{bug_rate_improvement:.0%}",
        "before_hours_per_pr": round(before_hours_per_pr, 2),
        "after_hours_per_pr": round(after_hours_per_pr, 2),
        "review_time_improvement": f"-{efficiency_improvement:.0%}",
        "total_api_cost_3months": f"${after.api_cost_usd}",
        "cost_per_pr": f"${after.api_cost_usd / after.total_prs:.4f}",
    }


result = calculate_improvement(before, after)
# {
#   "before_bug_rate_per_100pr": 3.85,
#   "after_bug_rate_per_100pr": 1.52,
#   "bug_rate_improvement": "-60%",
#   "before_hours_per_pr": 0.60,

# }
```

## 10.4 コストと削減効果の比較

```python
def roi_analysis() -> dict:
    """
    ROI（投資対効果）の計算.

    仮定:
    - エンジニアの時給: ¥5,000（年収800万円換算）
    - 本番バグ修正の工数: 平均4時間/件（調査2時間 + 修正1時間 + リリース1時間）
    - ステージングバグ修正: 平均1時間/件
    """
    ENGINEER_HOURLY_RATE_JPY = 5_000
    PROD_BUG_FIX_HOURS = 4
    STAGING_BUG_FIX_HOURS = 1
    USD_TO_JPY = 150

    # 削減されたバグ修正工数
    prod_bugs_reduced = 12 - 5  # 7件
    staging_bugs_reduced = 31 - 14  # 17件

    saved_hours = (
        prod_bugs_reduced * PROD_BUG_FIX_HOURS
        + staging_bugs_reduced * STAGING_BUG_FIX_HOURS
    )
    saved_cost_jpy = saved_hours * ENGINEER_HOURLY_RATE_JPY

    # レビュー工数の削減
    review_hours_saved = 186.0 - 142.0  # 44時間
    review_cost_saved_jpy = review_hours_saved * ENGINEER_HOURLY_RATE_JPY

    # APIコスト
    api_cost_jpy = int(3.82 * USD_TO_JPY)  # ¥573

    total_saved_jpy = saved_cost_jpy + review_cost_saved_jpy
    roi = (total_saved_jpy - api_cost_jpy) / api_cost_jpy

    return {
        "saved_bug_fix_hours": saved_hours,
        "saved_bug_fix_cost_jpy": f"¥{saved_cost_jpy:,}",
        "saved_review_hours": review_hours_saved,
        "saved_review_cost_jpy": f"¥{review_cost_saved_jpy:,}",
        "total_saved_jpy": f"¥{total_saved_jpy:,}",
        "api_cost_jpy": f"¥{api_cost_jpy:,}",
        "roi_ratio": f"{roi:.0f}x",  # 投資対効果
    }


roi = roi_analysis()
# {
#   "saved_bug_fix_hours": 45,
#   "saved_bug_fix_cost_jpy": "¥225,000",
#   "saved_review_hours": 44,
#   "saved_review_cost_jpy": "¥220,000",
#   "total_saved_jpy": "¥445,000",
#   "api_cost_jpy": "¥573",
#   "roi_ratio": "776x",   ← API費用の776倍の効果
# }
```

:::message
**ROIの解釈について**

776倍というROIは、エンジニア時給と本番バグ修正工数の仮定に強く依存しています。
ただし、「APIコストが月¥190程度で、バグ修正や工数削減の効果がある」という事実は揺るぎません。
費用対効果の優位性は高いですが、数字の誇張には注意してください。
:::

## 10.5 「AIレビューを入れてよかった」瞬間

### エピソード1: SQLインジェクション未遂

```python
# 実際にAIが発見したコード（匿名化）
class SearchHandler:
    def search_users(self, query_params: dict) -> list:
        name = query_params.get("name", "")
        # AIの指摘: "🔴 重大: SQLインジェクション脆弱性"
        sql = f"SELECT * FROM users WHERE name LIKE '%{name}%'"
        return self.db.execute(sql).fetchall()
```

このPRは「急ぎの検索機能追加」として出されたもので、レビュアーは2名いたが、ともにdiffの別の部分（ビジネスロジック）に集中していた。AIが先に発見し、エンジニアが修正してから人間レビューに回った。

レビュアーの一人のコメント：「AIがいなかったら確実に見落としてた。30分で実装した機能がSQLインジェクション持ちで本番に入るところだった。」

### エピソード2: 疲弊したレビュアーを支える

金曜夕方17:30に緊急のPRが出た（リリース前のバグ修正）。レビュアーは週末前の疲弊した状態だったが、AIが既に「問題なし（軽微な提案のみ）」と判定していた。

レビュアーは設計の確認のみ（5分）でマージを承認。「AIがいなかったら全部見直してたか、もしくは月曜まで待ってと言っていた」とレビュアーはレトロで話した。

## 10.6 「AIレビューで失敗した」事例

### 失敗1: AIを信頼しすぎてバグを見逃した

```python
# AIが「問題なし」と判定したコード
async def process_payments(payment_ids: list[int]) -> dict:
    results = {}
    for pid in payment_ids:
        payment = await get_payment(pid)
        if payment.status == "pending":
            await charge(payment)
            results[pid] = "charged"
    return results
```

AIはこのコードを「構文的に正しく、非同期処理も適切」と判定した。しかし本番で問題が発生した。**同じ支払いIDが複数回リストに入った場合、二重請求が発生する**バグだ。

これはビジネスロジックの問題であり、「べき等性（idempotency）」の概念をAIがコードから読み取れなかった。

**教訓**: AIが「問題なし」と言っても、ビジネスロジックに関わる部分は必ず人間がレビューする。

### 失敗2: AIの大量指摘でPRが埋まった

初期設定（プロンプトが粗かった時期）に、大量のdiffを持つリファクタリングPRを出したところ、AIが87件のコメントを投稿した。

開発者はコメントを全部読む気力を失い、「全部AIの指摘だから無視していいや」というモードになり始めた。重要なバグ指摘（SQLインジェクション1件）が埋もれてしまった。

**教訓**: 指摘は最大10件に絞る。重大な問題のみをコメントし、軽微な提案はサマリーにまとめる（第5章のプロンプト設計を参照）。

### 失敗3: AIが古いAPIを推奨した

```python
# AIの「改善提案」
# Before（開発者が書いたコード）:
async with httpx.AsyncClient() as client:
    response = await client.get(url)

# AIの提案: "aiohttp を使う方が非同期処理の効率が良いです。"
# → チームはhttpxに統一していた。切り替えはむしろ退行。
```

AIは訓練データの時点でaiohttpが広く使われていたため、プロジェクトのコンテキスト（httpx統一方針）を知らずに提案した。

**教訓**: AIの提案には必ず「チームのコンテキストと合っているか」の確認が必要。システムプロンプトにチームの技術選択を明記すると誤指摘が減る。

## 10.8 まとめ：AIコードレビューの正しい期待値設定

6ヶ月間のデータから言えることをまとめる。

```
AIコードレビューで期待できること:
  ✅ パターンが明確なバグ（SQL injection、未ハンドル例外等）の大幅減
  ✅ 本番バグ率を50〜60%削減（前提条件あり）
  ✅ 人間レビュー工数を20〜30%削減
  ✅ 月あたり数百円程度のコストで運用可能
  ✅ 疲弊・見落としによるバグの防止

AIコードレビューで期待できないこと:
  ❌ ビジネスロジックのバグを根絶する
  ❌ アーキテクチャ設計の問題を発見する
  ❌ UIバグを検出する
  ❌ 競合状態・スレッドセーフティ問題を検出する
  ❌ チームの文脈を理解した判断をする
  ❌ 人間レビューを不要にする

最終メッセージ:
  AIコードレビューは「Silver Bullet」ではない。
  しかし、低コストで明確なROIをもたらすツールだ。
  「パターンが明確な問題」に絞って使い、
  「文脈が必要な判断」は人間が担う。
  この分担を守れば、チームの品質は確実に向上する。
```

本書を通じて、単なる「Claude APIの使い方」ではなく、**AIと人間が協力してコード品質を上げるシステムの設計と運用**を伝えてきた。

あなたのチームでの実践を始める最初の一歩は、第3章のシンプルな実装から始めることだ。10行のPythonコードと1つのプロンプトが、チームの開発体験を変える出発点になる。
