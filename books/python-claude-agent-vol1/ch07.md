# 第7章: 評価・テスト・モニタリング：AIシステムの品質保証

## はじめに（なぜこれが重要か）

AIシステムの開発で最も見落とされやすい工程が「評価・テスト・モニタリング」です。

従来のソフトウェアテストは決定論的です。`add(2, 3)`が`5`を返すかどうかは、何万回実行しても変わりません。しかしLLMは確率的です。同じプロンプトを100回投げれば、100通りとは言わないまでも、品質にばらつきが生じます。

2023年に発表された論文「Evaluating Large Language Models: A Comprehensive Survey」（Chang et al., 2023）は、LLM評価の難しさを体系化しました。その核心は**「何を・どのように・なぜ評価するか」が曖昧なまま運用されている現場が多い**という指摘です。

```
LLM評価の3つの難題（Chang et al., 2023より）
1. 正解が一意でない（open-ended generation）
2. 評価者（人間）のバイアスと疲労
3. 分布シフト：本番データと評価データの乖離
```

この章では、これらの難題をnotecreatorの実装を通じて具体的に解決します。「動いているから大丈夫」から「計測しているから安心」への脱却が目標です。

## 7.1 LLM出力のユニットテスト：確率的テストの考え方

### アンチパターン：決定論的テストの失敗

まず、よくある失敗から見ます。

```python
# ❌ アンチパターン：決定論的テスト（必ず失敗する）
def test_article_generation():
    article = generate_article("Python入門")
    # LLMは毎回違う出力を返すため、完全一致は不可能
    assert article == "Pythonとは、1991年にグイド・ヴァン・ロッサムが..."  # 絶対に通らない
```

### 正しいアプローチ：プロパティベーステスト

LLMのテストは「特定の出力」ではなく「出力の性質（プロパティ）」を検証します。

```python
# tests/test_article_quality.py
import pytest
import re
import time
from unittest.mock import patch, MagicMock
from pathlib import Path
import sys

sys.path.insert(0, str(Path(__file__).parent.parent))


# ======================================================
# プロパティベーステスト：「何であるべきか」を検証する
# ======================================================

class ArticleProperties:
    """記事が満たすべき性質（プロパティ）の定義"""

    @staticmethod
    def has_minimum_length(text: str, minimum: int = 1200) -> bool:
        """最低文字数を満たすか"""
        return len(text.replace(" ", "").replace("\n", "")) >= minimum

    @staticmethod
    def has_no_markdown(text: str) -> bool:
        """プレーンテキストか（Markdown記法を含まないか）"""
        markdown_patterns = [
            r"^#{1,6}\s",      # 見出し
            r"\*\*[^*]+\*\*",  # 太字
            r"^```",           # コードブロック
            r"^\- |\^\* ",     # 箇条書き
        ]
        for pattern in markdown_patterns:
            if re.search(pattern, text, re.MULTILINE):
                return False
        return True

    @staticmethod
    def has_no_ng_expressions(text: str, ng_list: list) -> bool:
        """NG表現を含まないか"""
        return not any(ng in text for ng in ng_list)

    @staticmethod
    def has_pr_notation(text: str) -> bool:
        """PR表記が冒頭200文字以内にあるか"""
        return "※本記事には広告が含まれます" in text[:200]

    @staticmethod
    def has_required_sections(text: str, sections: list) -> tuple[bool, list]:
        """必須セクションが存在するか"""
        missing = [s for s in sections if s not in text]
        return len(missing) == 0, missing


# ======================================================
# モンキーパッチによる確率的テストの安定化
# ======================================================

SAMPLE_VALID_ARTICLE = """※本記事には広告が含まれます

この記事でわかること

Pythonでのデータ分析の基礎について解説します。
""" + "テスト用のコンテンツです。" * 80 + """

この記事で紹介したツール

NumPy、pandas、matplotlibを紹介しました。

おすすめのサービス

Udemyのデータサイエンスコースをおすすめします。

さらに学びたい方へ

公式ドキュメントを参照してください。
"""

SAMPLE_INVALID_ARTICLE = """# Pythonデータ分析入門

**絶対に**稼げる方法を紹介します！
"""


class TestArticleProperties:
    """プロパティベーステスト群"""

    def test_valid_article_passes_all_checks(self):
        """正常な記事はすべてのプロパティを満たす"""
        props = ArticleProperties()
        ng_list = ["絶対", "確実に稼げる", "100%"]

        assert props.has_minimum_length(SAMPLE_VALID_ARTICLE, 1200), "文字数不足"
        assert props.has_pr_notation(SAMPLE_VALID_ARTICLE), "PR表記なし"
        assert props.has_no_ng_expressions(SAMPLE_VALID_ARTICLE, ng_list), "NG表現あり"

    def test_invalid_article_fails_markdown_check(self):
        """Markdownを含む記事は検出される"""
        props = ArticleProperties()
        assert not props.has_no_markdown(SAMPLE_INVALID_ARTICLE)

    def test_invalid_article_fails_ng_expression_check(self):
        """NG表現を含む記事は検出される"""
        props = ArticleProperties()
        ng_list = ["絶対", "確実に稼げる", "100%"]
        assert not props.has_no_ng_expressions(SAMPLE_INVALID_ARTICLE, ng_list)

    def test_required_sections_detection(self):
        """必須セクション不足が検出される"""
        props = ArticleProperties()
        sections = ["この記事でわかること", "この記事で紹介したツール",
                    "おすすめのサービス", "さらに学びたい方へ"]
        passed, missing = props.has_required_sections(SAMPLE_VALID_ARTICLE, sections)
        assert passed, f"不足セクション: {missing}"


# ======================================================
# 統計的テスト：N回実行して合格率を検証する
# ======================================================

class TestStochasticBehavior:
    """確率的テスト：合格率が閾値を超えるか"""

    def _mock_generate(self, fail_rate: float = 0.1) -> str:
        """テスト用のモック生成関数（fail_rateの確率で不正記事を返す）"""
        import random
        if random.random() < fail_rate:
            return SAMPLE_INVALID_ARTICLE
        return SAMPLE_VALID_ARTICLE

    def test_pass_rate_above_threshold(self):
        """
        本番品質基準: N回実行中、90%以上が品質チェックを通過すること
        
        なぜこのテストが必要か：
        - LLMは確率的なので1回の成功は信頼できない
        - 合格率を統計的に保証することで品質のSLAを設定できる
        """
        N = 30  # テスト実行回数（本番では100以上推奨）
        THRESHOLD = 0.90  # 合格率閾値

        ng_list = ["絶対", "確実に稼げる", "100%"]
        pass_count = 0

        for _ in range(N):
            article = self._mock_generate(fail_rate=0.05)  # 5%の確率で失敗
            props = ArticleProperties()

            # すべてのプロパティを満たす場合のみパス
            if (props.has_minimum_length(article) and
                    props.has_no_ng_expressions(article, ng_list) and
                    props.has_pr_notation(article)):
                pass_count += 1

        pass_rate = pass_count / N
        assert pass_rate >= THRESHOLD, (
            f"合格率 {pass_rate:.1%} が閾値 {THRESHOLD:.1%} を下回りました。"
            f"プロンプトの見直しが必要です。"
        )
```

このテスト設計の核心は「**合格率のSLA（Service Level Agreement）を設定する**」という発想です。LLMは確率機械なので「必ず100点を取れ」ではなく「90%以上の確率で合格点を取れ」という基準が現実的です。

## 7.2 LLM-as-Judge：AIがAIを評価するフレームワーク

人間が毎回LLMの出力を評価するのはスケールしません。Chang et al.（2023）が注目した手法が「LLM-as-Judge」——別のLLMが評価者になるアプローチです。

### なぜLLM-as-Judgeが有効か

| 評価手法 | コスト | スケール | 一貫性 |
|---|---|---|---|
| 人間評価 | 高 | 低 | ばらつき大 |
| ルールベース | 低 | 高 | 硬直 |
| LLM-as-Judge | 中 | 高 | 高（同じモデルなら一貫） |

```python
# automation/quality/llm_judge.py
"""LLM-as-Judge: ClaudeでClaudeの出力を評価する"""

import json
import re
import time
from dataclasses import dataclass, field
from typing import Optional
import anthropic


@dataclass
class JudgementResult:
    """評価結果を構造化データとして保持"""
    overall_score: float          # 0.0 - 1.0
    readability_score: float      # 読みやすさ
    accuracy_score: float         # 正確性（事実の正確さ）
    engagement_score: float       # 読者の興味を引くか
    compliance_score: float       # ガイドライン準拠
    reasoning: str                # 評価理由（重要：なぜその点数か）
    improvement_suggestions: list[str] = field(default_factory=list)
    raw_response: str = ""


JUDGE_PROMPT_TEMPLATE = """あなたは記事品質の専門評価者です。
以下の記事を厳密に評価し、必ずJSON形式のみで回答してください。

【評価対象記事】
---
{article}
---

【評価基準】
以下の4軸で0.0〜1.0のスコアをつけてください：

1. readability（読みやすさ）
   - 1.0: 流れが自然で読み疲れしない
   - 0.5: 一部わかりにくい箇所がある
   - 0.0: 構成が乱れ理解しにくい

2. accuracy（正確性）
   - 1.0: 事実・数値に誤りがない
   - 0.5: 軽微な不正確さがある
   - 0.0: 明らかな誤情報を含む

3. engagement（エンゲージメント）
   - 1.0: 続きを読みたくなる
   - 0.5: 普通
   - 0.0: 退屈で離脱したくなる

4. compliance（ガイドライン準拠）
   - 1.0: 広告表記・構成すべて準拠
   - 0.5: 軽微な不備がある
   - 0.0: 重大な違反がある

【必須出力形式】
```json
{{
  "readability_score": 0.0,
  "accuracy_score": 0.0,
  "engagement_score": 0.0,
  "compliance_score": 0.0,
  "overall_score": 0.0,
  "reasoning": "評価理由を200字以内で",
  "improvement_suggestions": ["改善点1", "改善点2"]
}}
```
JSONのみ出力し、それ以外のテキストは一切含めないでください。"""


class LLMJudge:
    """Claude を使った記事品質評価エンジン"""

    def __init__(self, model: str = "claude-opus-4-5"):
        self.client = anthropic.Anthropic()
        self.model = model
        # judge用は軽量モデルでコストを抑える
        self.judge_model = "claude-haiku-4-5"

    def evaluate(self, article: str, max_retries: int = 3) -> JudgementResult:
        """
        記事を評価してJudgementResultを返す

        コスト試算:
        - claude-haiku-4-5: 入力$0.25/MTok, 出力$1.25/MTok
        - 記事1本(約2000字)の評価: 入力~1000tok + 出力~300tok ≈ $0.00063
        - 月1000本評価: 約$0.63（ほぼ無視できるコスト）
        """
        prompt = JUDGE_PROMPT_TEMPLATE.format(article=article[:3000])  # 長すぎる場合は切り詰め

        for attempt in range(max_retries):
            try:
                response = self.client.messages.create(
                    model=self.judge_model,
                    max_tokens=512,
                    messages=[{"role": "user", "content": prompt}]
                )
                raw_text = response.content[0].text

                # JSONのパースを試みる（コードブロックを除去）
                json_text = re.sub(r"```json\s*|\s*```", "", raw_text).strip()
                data = json.loads(json_text)

                # overall_scoreが未設定の場合は4軸の平均を計算
                if "overall_score" not in data or data["overall_score"] == 0.0:
                    data["overall_score"] = (
                        data.get("readability_score", 0) +
                        data.get("accuracy_score", 0) +
                        data.get("engagement_score", 0) +
                        data.get("compliance_score", 0)
                    ) / 4

                return JudgementResult(
                    overall_score=data["overall_score"],
                    readability_score=data["readability_score"],
                    accuracy_score=data["accuracy_score"],
                    engagement_score=data["engagement_score"],
                    compliance_score=data["compliance_score"],
                    reasoning=data.get("reasoning", ""),
                    improvement_suggestions=data.get("improvement_suggestions", []),
                    raw_response=raw_text,
                )

            except (json.JSONDecodeError, KeyError) as e:
                if attempt < max_retries - 1:
                    time.sleep(2 ** attempt)  # 指数バックオフ
                    continue
                # 最終試行も失敗した場合は最低スコアで記録（サイレント失敗を防ぐ）
                return JudgementResult(
                    overall_score=0.0,
                    readability_score=0.0,
                    accuracy_score=0.0,
                    engagement_score=0.0,
                    compliance_score=0.0,
                    reasoning=f"評価エラー: {str(e)}",
                    raw_response=raw_text if 'raw_text' in locals() else "",
                )

    def batch_evaluate(self, articles: list[str]) -> list[JudgementResult]:
        """複数記事の一括評価（レート制限を考慮）"""
        results = []
        for i, article in enumerate(articles):
            result = self.evaluate(article)
            results.append(result)
            # APIレート制限対策: 評価間に1秒のインターバル
            if i < len(articles) - 1:
                time.sleep(1)
        return results


# 動作確認
if __name__ == "__main__":
    sample = """※本記事には広告が含まれます

    この記事でわかること

    Pythonを使ったデータ分析の基礎を学べます。初心者の方でも
    ステップバイステップで理解できるように解説しています。
    """ + "具体的な解説内容。" * 60

    judge = LLMJudge()
    result = judge.evaluate(sample)
    print(f"総合スコア: {result.overall_score:.2f}")
    print(f"読みやすさ: {result.readability_score:.2f}")
    print(f"評価理由: {result.reasoning}")
    print(f"改善提案: {result.improvement_suggestions}")
```
