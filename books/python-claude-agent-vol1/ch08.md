# 第8章: AIエージェントアーキテクチャ大全：ReAct・RAG・マルチエージェント

## はじめに（なぜこれが重要か）

「エージェントを作った。でも本番で暴走した」

これは2024年時点で最も多く寄せられる相談の一つです。Claude APIを使えばエージェントは動く。しかし、**なぜそのアーキテクチャを選ぶのか**を理解していないと、コストが10倍になり、ループが止まらず、出力品質がガタ落ちします。

本章では、研究論文に裏打ちされた3つの主要パターン——ReAct系変種・RAG・マルチエージェント——を**動くコードとコスト数値付きで**解説します。冒頭に示したnotecreator（AI記事工場）の実装は、これらパターンを実際の業務自動化に組み合わせた好例です。その設計思想を読み解きながら、あなたの現場にそのまま持ち込める知識を手に入れましょう。

### 本章で扱う研究

| 論文 | 年 | 核心アイデア | 実務での意味 |
|---|---|---|---|
| ReAct (Yao et al.) | 2022 | 推論とアクションを交互に実行 | ツール呼び出しの基本設計 |
| Reflexion (Shinn et al.) | 2023 | 失敗を言語で記憶し自己修正 | リトライ品質の向上 |
| CoALA (Sumers et al.) | 2023 | 認知アーキテクチャの統一モデル | メモリ設計の体系化 |

## 8.1 ReAct変種の比較実装：PlanAct・Reflexion・ReWOO

### ReActの本質を理解する

ReAct論文の核心は「思考(Thought) → 行動(Action) → 観察(Observation)」のループです。しかし実務では**3つの変種**が用途によって使い分けられます。

```
ReAct:      Thought → Action → Obs → Thought → Action → Obs ...（逐次）
PlanAct:    Plan全体を立案 → Actions並列実行（並列化向き）
ReWOO:      計画を先に全部立てる → ツール実行 → まとめて推論（コスト最小化）
Reflexion:  ReAct + 失敗時の自己評価ループ（品質重視）
```

以下に4パターンを統一インターフェースで実装します。

```python
# agent_architectures.py
"""
ReAct系エージェント4パターン完全実装
依存: pip install anthropic tiktoken
"""

import json
import time
from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Any, Callable

import anthropic

# ─────────────────────────────────────────────
# 共通データ構造
# ─────────────────────────────────────────────

@dataclass
class ToolResult:
    tool_name: str
    output: str
    success: bool
    error: str = ""

@dataclass
class AgentTrace:
    """実行トレースと統計（コスト管理に必須）"""
    thoughts: list[str] = field(default_factory=list)
    actions: list[str] = field(default_factory=list)
    observations: list[str] = field(default_factory=list)
    reflections: list[str] = field(default_factory=list)
    input_tokens: int = 0
    output_tokens: int = 0

    @property
    def estimated_cost_usd(self) -> float:
        # Claude Sonnet 4 の価格（2024年時点）
        return (self.input_tokens * 3.0 + self.output_tokens * 15.0) / 1_000_000

    def summary(self) -> str:
        return (
            f"ステップ数: {len(self.actions)} | "
            f"トークン: in={self.input_tokens} out={self.output_tokens} | "
            f"推定コスト: ${self.estimated_cost_usd:.4f}"
        )


# ─────────────────────────────────────────────
# ツール定義（全パターン共通）
# ─────────────────────────────────────────────

def make_tools() -> dict[str, Callable]:
    """サンプルツール集（本番では外部API呼び出しに差し替え）"""

    def web_search(query: str) -> str:
        # 本番: Tavily/Brave Search API を呼ぶ
        return f"[検索結果] '{query}' に関する情報: Python 3.12がリリースされ高速化した。"

    def read_file(path: str) -> str:
        try:
            with open(path) as f:
                return f.read()[:2000]  # 長すぎるファイルは切り詰め
        except FileNotFoundError:
            return f"ERROR: ファイル {path} が見つかりません"

    def write_file(path: str, content: str) -> str:
        with open(path, "w") as f:
            f.write(content)
        return f"OK: {path} に {len(content)} 文字を書き込みました"

    def calculate(expression: str) -> str:
        try:
            # 安全な評価（evalは絶対に使わない！）
            allowed = {k: v for k, v in __import__("math").__dict__.items()
                       if not k.startswith("_")}
            result = eval(compile(expression, "<calc>", "eval"),
                          {"__builtins__": {}}, allowed)
            return str(result)
        except Exception as e:
            return f"ERROR: {e}"

    return {
        "web_search": web_search,
        "read_file": read_file,
        "write_file": write_file,
        "calculate": calculate,
    }


# ─────────────────────────────────────────────
# パターン1: 標準ReAct
# ─────────────────────────────────────────────

class ReActAgent:
    """
    論文準拠のReActエージェント。
    思考→行動→観察を繰り返す最もシンプルな実装。
    用途: 汎用タスク、デバッグしやすさ重視
    """

    SYSTEM_PROMPT = """あなたはReActエージェントです。以下の形式を厳守してください:

Thought: タスクについて考える
Action: tool_name({"key": "value"}) ← 必ずJSON形式
Observation: [ツールの結果がここに入る]
... (繰り返し)
Final Answer: 最終回答

利用可能ツール: web_search, read_file, write_file, calculate
"""

    def __init__(self, client: anthropic.Anthropic, model: str = "claude-sonnet-4-5"):
        self.client = client
        self.model = model
        self.tools = make_tools()

    def run(self, task: str, max_steps: int = 10) -> tuple[str, AgentTrace]:
        trace = AgentTrace()
        messages = [{"role": "user", "content": task}]

        for step in range(max_steps):
            response = self.client.messages.create(
                model=self.model,
                max_tokens=1024,
                system=self.SYSTEM_PROMPT,
                messages=messages,
            )
            trace.input_tokens += response.usage.input_tokens
            trace.output_tokens += response.usage.output_tokens

            text = response.content[0].text
            messages.append({"role": "assistant", "content": text})

            # Final Answer で終了
            if "Final Answer:" in text:
                answer = text.split("Final Answer:")[-1].strip()
                return answer, trace

            # Actionを抽出して実行
            if "Action:" in text:
                action_line = [l for l in text.split("\n") if l.startswith("Action:")][0]
                tool_result = self._execute_action(action_line)
                trace.actions.append(action_line)
                trace.observations.append(tool_result.output)

                # ObservationをAssistantメッセージの後に追加
                messages.append({
                    "role": "user",
                    "content": f"Observation: {tool_result.output}"
                })

        return "最大ステップ数に達しました", trace

    def _execute_action(self, action_line: str) -> ToolResult:
        """Action: tool_name({"key": "value"}) をパースして実行"""
        try:
            # "Action: " プレフィックスを除去
            action_str = action_line.replace("Action:", "").strip()
            # tool_name(json) をパース
            paren_pos = action_str.index("(")
            tool_name = action_str[:paren_pos].strip()
            args_str = action_str[paren_pos + 1:-1]  # 括弧の中身
            args = json.loads(args_str)

            if tool_name not in self.tools:
                return ToolResult(tool_name, "", False, f"ツール '{tool_name}' は存在しません")

            # ツール実行
            output = self.tools[tool_name](**args)
            return ToolResult(tool_name, output, True)

        except Exception as e:
            return ToolResult("unknown", "", False, f"パースエラー: {e}")

# パターン2: Reflexion（自己修正ループ）
# ─────────────────────────────────────────────

class ReflexionAgent:
    """
    Reflexion論文準拠。失敗を「言語的記憶」として蓄積し再試行。
    用途: 品質重視タスク（記事生成・コードレビュー等）
    コスト: ReActの2-3倍だが品質が大幅向上

    notecreatorのinspect_and_save()がこのパターンに相当する。
    品質チェック → 失敗ならフィードバック付きで再生成。
    """

    def __init__(self, client: anthropic.Anthropic, model: str = "claude-sonnet-4-5"):
        self.client = client
        self.model = model
        self.memory: list[str] = []  # 言語的長期記憶

    def run(
        self,
        task: str,
        evaluator: Callable[[str], tuple[bool, str]],
        max_trials: int = 3,
    ) -> tuple[str, AgentTrace]:
        """
        evaluator: (output) -> (成功かどうか, フィードバック文字列)
        """
        trace = AgentTrace()
        react_agent = ReActAgent(self.client, self.model)

        for trial in range(max_trials):
            # 過去の失敗記憶をタスクに注入
            augmented_task = task
            if self.memory:
                memory_str = "\n".join(f"- {m}" for m in self.memory)
                augmented_task = (
                    f"{task}\n\n"
                    f"【過去の失敗から学んだこと】\n{memory_str}\n"
                    f"上記の失敗を繰り返さないよう注意して実行してください。"
                )

            output, step_trace = react_agent.run(augmented_task)
            trace.input_tokens += step_trace.input_tokens
            trace.output_tokens += step_trace.output_tokens

            # 評価
            success, feedback = evaluator(output)
            if success:
                trace.reflections.append(f"Trial {trial+1}: 成功")
                return output, trace

            # 失敗: 反省を言語化して記憶
            reflection = self._reflect(task, output, feedback, trace)
            self.memory.append(reflection)
            trace.reflections.append(f"Trial {trial+1}: 失敗 → {reflection}")

        return output, trace  # 最後の出力を返す

    def _reflect(self, task: str, output: str, feedback: str, trace: AgentTrace) -> str:
        """失敗を分析して次回への教訓を生成"""
        prompt = f"""タスク: {task}

あなたの出力: {output}

評価フィードバック: {feedback}

この失敗から学べる具体的な教訓を1文で述べてください。
次回同じミスを犯さないための行動指針として記述してください。"""

        response = self.client.messages.create(
            model=self.model,
            max_tokens=256,
            messages=[{"role": "user", "content": prompt}],
        )
        trace.input_tokens += response.usage.input_tokens
        trace.output_tokens += response.usage.output_tokens
        return response.content[0].text.strip()

# パターン3: ReWOO（コスト最小化）
# ─────────────────────────────────────────────

class ReWOOAgent:
    """
    ReWOO: Reasoning WithOut Observation（Xu et al. 2023）
    先に全アクション計画を立て、観察なしで並列実行→まとめて推論。
    用途: 独立したツール呼び出しが多い場合。コストをReActの50-70%に削減。
    制限: 前ステップの結果を見てツールを選ぶ必要がある場合は使えない。
    """

    def __init__(self, client: anthropic.Anthropic, model: str = "claude-sonnet-4-5"):
        self.client = client
        self.model = model
        self.tools = make_tools()

    def run(self, task: str) -> tuple[str, AgentTrace]:
        trace = AgentTrace()

        # フェーズ1: 計画立案（ツール結果を見ずに全計画を作成）
        plan = self._make_plan(task, trace)

        # フェーズ2: 並列実行（本番はasyncio.gather推奨）
        evidences = {}
        for i, (tool_name, args, var_name) in enumerate(plan):
            if tool_name in self.tools:
                result = self.tools[tool_name](**args)
                evidences[var_name] = result
                trace.actions.append(f"{var_name} = {tool_name}({args})")
                trace.observations.append(f"{var_name} = {result[:100]}...")

        # フェーズ3: まとめて推論
        answer = self._solve(task, evidences, trace)
        return answer, trace

    def _make_plan(
        self, task: str, trace: AgentTrace
    ) -> list[tuple[str, dict, str]]:
        prompt = f"""タスク: {task}

利用可能ツール: web_search(query), calculate(expression), read_file(path)

このタスクを解くために必要なツール呼び出しを計画してください。
各行を以下の形式で出力:
#E1 = web_search({{"query": "検索ワード"}})
#E2 = calculate({{"expression": "1+1"}})

JSONは必ず二重引用符を使うこと。計画のみ出力し、説明不要。"""

        response = self.client.messages.create(
            model=self.model,
            max_tokens=512,
            messages=[{"role": "user", "content": prompt}],
        )
        trace.input_tokens += response.usage.input_tokens
        trace.output_tokens += response.usage.output_tokens

        return self._parse_plan(response.content[0].text)

    def _parse_plan(self, text: str) -> list[tuple[str, dict, str]]:
        plan = []
        for line in text.strip().split("\n"):
            if not line.startswith("#"):
                continue
            try:
                var_name, rest = line.split("=", 1)
                var_name = var_name.strip()
                rest = rest.strip()
                paren_pos = rest.index("(")
                tool_name = rest[:paren_pos].strip()
                args = json.loads(rest[paren_pos + 1:-1])
                plan.append((tool_name, args, var_name))
            except Exception:
                continue
        return plan

    def _solve(
        self, task: str, evidences: dict[str, str], trace: AgentTrace
    ) -> str:
        evidence_str = "\n".join(f"{k}: {v}" for k, v in evidences.items())
        prompt = f"""タスク: {task}

収集した情報:
{evidence_str}

上記の情報を使ってタスクに答えてください。"""

        response = self.client.messages.create(
            model=self.model,
            max_tokens=1024,
            messages=[{"role": "user", "content": prompt}],
        )
        trace.input_tokens += response.usage.input_tokens
        trace.output_tokens += response.usage.output_tokens
        return response.content[0].text.strip()
